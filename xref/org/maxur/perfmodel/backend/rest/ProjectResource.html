44  
45      private static final String VERSION = "version";
46  
47      private static final String ID = "id";
48  
49      private static final String PERFORMANCE_MODEL_IS_NOT_FOUNDED = "Performance Model '%s' is not founded";

51      private static final String PERFORMANCE_MODEL_IS_NOT_SAVED = "Performance Model is not saved";
52  
53      @Inject
54      private Repository<Project> repository;
55  
56  
57      @GET
58      @Produces(MediaType.APPLICATION_JSON)
59      public Collection<Project> findAll() {
60          return repository.findAll();
61      }
62  
63      @GET
64      @Path("/{name}")
65      @Produces(MediaType.APPLICATION_JSON)
66      public String load(@PathParam(NAME) String name) {
67          final Project project = repository.get(name);
68          if (project == null) {
69              LOGGER.error(format(PERFORMANCE_MODEL_IS_NOT_FOUNDED, name));
70              throw notFoundException(format(PERFORMANCE_MODEL_IS_NOT_FOUNDED, name));
71          }
72          return project.asRaw();
73      }
74  
75      @DELETE
76      @Path("/{name}")
77      @Produces(MediaType.APPLICATION_JSON)
78      public Project delete(@PathParam(NAME) String name) {
79          final Project project = repository.remove(name);
80          if (project == null) {
81              LOGGER.error(format(PERFORMANCE_MODEL_IS_NOT_FOUNDED, name));
82              throw notFoundException(format(PERFORMANCE_MODEL_IS_NOT_FOUNDED, name));
83          }
84          return  project;
85      }
86  
87      @POST
88      @Consumes(MediaType.APPLICATION_JSON)
89      @Produces(MediaType.APPLICATION_JSON)
90      public synchronized Response save(final String object) {
91          try {
92              final ObjectMapper mapper = new ObjectMapper(new JsonFactory());
93              final Map<String,Object> map = mapper.readValue(object, new TypeReference<HashMap<String,Object>>(){});
94  
95              final String id = (String) map.get(ID);
96              final String name = (String) map.get(NAME);
97              final Integer version = (Integer) map.get(VERSION);
98              final Project previousProjectVersion = repository.get(name);
99              final Project project;
100             if (previousProjectVersion != null) {
101                 project = new Project(id, name, version + 1);
102                 project.checkConflictWith(previousProjectVersion);
103             } else {
104                 project = new Project(id, name, 1);
105             }
106             map.put(VERSION, project.getVersion());
107             project.setRaw(mapper.writeValueAsString(map));
108             repository.put(project);
109             return Response.status(Response.Status.CREATED).entity(project).build();
110         } catch (ValidationException e) {
111             LOGGER.debug(PERFORMANCE_MODEL_IS_NOT_SAVED, e);
112             LOGGER.info(PERFORMANCE_MODEL_IS_NOT_SAVED, e.getMessage());
113             throw conflictException(PERFORMANCE_MODEL_IS_NOT_SAVED, e.getMessage());
114         } catch (IOException | NumberFormatException e) {
115             LOGGER.debug(PERFORMANCE_MODEL_IS_NOT_SAVED, e);
116             LOGGER.error(PERFORMANCE_MODEL_IS_NOT_SAVED, e);
117             throw badRequestException(PERFORMANCE_MODEL_IS_NOT_SAVED, e.getMessage());
118         }
119     }
