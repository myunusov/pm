16  package org.maxur.perfmodel.backend.domain;
17  
18  import javax.xml.bind.annotation.XmlRootElement;
19  import java.io.Serializable;
20  
21  /**
22   * @author Maxim Yunusov
23   * @version 1.0 24.11.13
24   */
25  @XmlRootElement
26  public class Project implements Serializable {
27  
28      private static final long serialVersionUID = -5236202133124299315L;
29  
30      private String raw;
31  
32      private String id;
33  
34      private String name;
35  
36      private int version;
37  
38      @SuppressWarnings("UnusedDeclaration")
39      public Project() {
40      }
41  
42      public Project(final String id, final String name, final int version) {
43          this.id = id;
44          this.name = name;
45          this.version = version;
46      }
47  
48  
49      public void checkConflictWith(final Project previousProjectVersion) throws ValidationException {
50          if (previousProjectVersion == null) {
51              return;
52          }
53          if (!id.equals(previousProjectVersion.getId())) {
54              throw new ValidationException(String.format("Performance Model '%s' already exists.", name));
55          }
56          if (version - 1 != previousProjectVersion.getVersion()) {
57              throw new ValidationException(
58                      String.format("Performance Model '%s' has already been changed by another user.", name)
59              );
60          }
61      }
62  
63      public String getName() {
64          return name;
65      }
66  
67      public String getId() {
68          return id;
69      }
70  
71      public int getVersion() {
72          return version;
73      }
74  
75      public void setRaw(String raw) {
76          this.raw = raw;
77      }
78  
79      public String asRaw() {
80          return raw;
81      }
82  }
