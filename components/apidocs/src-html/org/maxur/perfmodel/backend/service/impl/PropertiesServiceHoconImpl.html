001package org.maxur.perfmodel.backend.service.impl;
002
003import com.typesafe.config.Config;
004import com.typesafe.config.ConfigException;
005import com.typesafe.config.ConfigFactory;
006import org.jvnet.hk2.annotations.Service;
007import org.maxur.perfmodel.backend.service.PropertiesService;
008import org.slf4j.Logger;
009import org.slf4j.LoggerFactory;
010
011import javax.annotation.PostConstruct;
012
013/**
014 * @author Maxim Yunusov
015 * @version 1.0
016 * @since <pre>9/2/2015</pre>
017 */
018@Service
019public class PropertiesServiceHoconImpl implements PropertiesService {
020
021    private static final Logger LOGGER = LoggerFactory.getLogger(PropertiesServiceHoconImpl.class);
022
023    private Config defaultConfig;
024
025    private Config userConfig;
026
027    @PostConstruct
028    public void init() {
029        defaultConfig = ConfigFactory.load().getConfig("DEFAULTS");
030        userConfig = ConfigFactory.load().getConfig("PMC");
031    }
032
033
034    @Override
035    public String asString(final String key) {
036        String value;
037        try {
038            try {
039                value = userConfig.getString(key);
040            } catch (ConfigException.Missing e) {
041                value = defaultConfig.getString(key);
042            }
043        } catch (ConfigException.Missing e) {
044            LOGGER.error("Configuration parameter '{}' is not found.", key);
045            throw e;
046        }
047        LOGGER.info("Configuration parameter {} = '{}'", key, value);
048        return value;
049    }
050
051
052
053}
