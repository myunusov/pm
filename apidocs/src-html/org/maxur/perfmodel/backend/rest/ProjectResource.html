001package org.maxur.perfmodel.backend.rest;
002
003
004import com.fasterxml.jackson.core.JsonFactory;
005import com.fasterxml.jackson.core.type.TypeReference;
006import com.fasterxml.jackson.databind.ObjectMapper;
007import org.maxur.perfmodel.backend.domain.Project;
008import org.maxur.perfmodel.backend.domain.Repository;
009import org.maxur.perfmodel.backend.domain.ValidationException;
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012
013import javax.inject.Inject;
014import javax.ws.rs.Consumes;
015import javax.ws.rs.DELETE;
016import javax.ws.rs.GET;
017import javax.ws.rs.POST;
018import javax.ws.rs.Path;
019import javax.ws.rs.PathParam;
020import javax.ws.rs.Produces;
021import javax.ws.rs.core.MediaType;
022import javax.ws.rs.core.Response;
023import java.io.IOException;
024import java.util.Collection;
025import java.util.HashMap;
026import java.util.Map;
027
028import static java.lang.String.format;
029import static org.maxur.perfmodel.backend.rest.WebException.badRequestException;
030import static org.maxur.perfmodel.backend.rest.WebException.conflictException;
031import static org.maxur.perfmodel.backend.rest.WebException.notFoundException;
032
033
034/**
035 * @author Maxim Yunusov
036 * @version 1.0 01.09.13
037 */
038@Path("/{a:projects}")
039public class ProjectResource {
040
041    private static final Logger LOGGER = LoggerFactory.getLogger(ProjectResource.class);
042
043    private static final String NAME = "name";
044
045    private static final String VERSION = "version";
046
047    private static final String ID = "id";
048
049    private static final String PERFORMANCE_MODEL_IS_NOT_FOUNDED = "Performance Model '%s' is not founded";
050
051    private static final String PERFORMANCE_MODEL_IS_NOT_SAVED = "Performance Model is not saved";
052
053    @Inject
054    private Repository<Project> repository;
055
056
057    @GET
058    @Produces(MediaType.APPLICATION_JSON)
059    public Collection<Project> findAll() {
060        return repository.findAll();
061    }
062
063    @GET
064    @Path("/{name}")
065    @Produces(MediaType.APPLICATION_JSON)
066    public String load(@PathParam(NAME) String name) {
067        final Project project = repository.get(name);
068        if (project == null) {
069            LOGGER.error(format(PERFORMANCE_MODEL_IS_NOT_FOUNDED, name));
070            throw notFoundException(format(PERFORMANCE_MODEL_IS_NOT_FOUNDED, name));
071        }
072        return project.asRaw();
073    }
074
075    @DELETE
076    @Path("/{name}")
077    @Produces(MediaType.APPLICATION_JSON)
078    public Project delete(@PathParam(NAME) String name) {
079        final Project project = repository.remove(name);
080        if (project == null) {
081            LOGGER.error(format(PERFORMANCE_MODEL_IS_NOT_FOUNDED, name));
082            throw notFoundException(format(PERFORMANCE_MODEL_IS_NOT_FOUNDED, name));
083        }
084        return  project;
085    }
086
087    @POST
088    @Consumes(MediaType.APPLICATION_JSON)
089    @Produces(MediaType.APPLICATION_JSON)
090    public synchronized Response save(final String object) {
091        try {
092            final ObjectMapper mapper = new ObjectMapper(new JsonFactory());
093            final Map<String,Object> map = mapper.readValue(object, new TypeReference<HashMap<String,Object>>(){});
094
095            final String id = (String) map.get(ID);
096            final String name = (String) map.get(NAME);
097            final Integer version = (Integer) map.get(VERSION);
098            final Project previousProjectVersion = repository.get(name);
099            final Project project;
100            if (previousProjectVersion != null) {
101                project = new Project(id, name, version + 1);
102                project.checkConflictWith(previousProjectVersion);
103            } else {
104                project = new Project(id, name, 1);
105            }
106            map.put(VERSION, project.getVersion());
107            project.setRaw(mapper.writeValueAsString(map));
108            repository.put(project);
109            return Response.status(Response.Status.CREATED).entity(project).build();
110        } catch (ValidationException e) {
111            LOGGER.info(PERFORMANCE_MODEL_IS_NOT_SAVED, e.getMessage());
112            throw conflictException(PERFORMANCE_MODEL_IS_NOT_SAVED, e.getMessage());
113        } catch (IOException | NumberFormatException e) {
114            LOGGER.error(PERFORMANCE_MODEL_IS_NOT_SAVED, e);
115            throw badRequestException(PERFORMANCE_MODEL_IS_NOT_SAVED, e.getMessage());
116        }
117    }
118
119
120}
