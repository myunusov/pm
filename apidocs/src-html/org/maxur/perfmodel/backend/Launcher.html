016package org.maxur.perfmodel.backend;
017
018import org.glassfish.hk2.api.TypeLiteral;
019import org.glassfish.hk2.utilities.binding.AbstractBinder;
020import org.glassfish.jersey.jackson.JacksonFeature;
021import org.glassfish.jersey.server.ResourceConfig;
022import org.maxur.perfmodel.backend.domain.Project;
023import org.maxur.perfmodel.backend.domain.Repository;
024import org.maxur.perfmodel.backend.infrastructure.PropertiesService;
025import org.maxur.perfmodel.backend.infrastructure.SimpleFileRepository;
026import org.maxur.perfmodel.backend.infrastructure.WebServer;
027import org.maxur.perfmodel.backend.rest.PMObjectMapperProvider;
028import org.slf4j.Logger;
029
030import static org.slf4j.LoggerFactory.getLogger;
031
032/**
033 * @author Maxim Yunusov
034 * @version 1.0 14.09.2014
035 */
036public class Launcher {
037
038    private static final Logger LOGGER = getLogger(Launcher.class);
039
040    private WebServer webServer;
041
042    private PropertiesService propertiesService;
043
044    public static void main(String[] args) throws Exception {
045        final Launcher client = new Launcher();
046        client.init();
047        client.run();
048    }
049
050    private void init() {
051        try {
052            propertiesService = PropertiesService.make("/pm.properties");
053            webServer = new WebServer(propertiesService);
054            webServer.start(createApp());
055        } catch (RuntimeException e) {
056            LOGGER.error("System don't initialising", e);
057        }
058    }
059
060    public ResourceConfig createApp() {
061        return new ResourceConfig() {
062            {
063                packages("org.maxur.perfmodel.backend.rest");
064                register(JacksonFeature.class);
065                register(PMObjectMapperProvider.class);
066                register(new AbstractBinder() {
067                    @Override
068                    protected void configure() {
069                        bind(propertiesService).to(PropertiesService.class);
070                        bindAsContract(new TypeLiteral<SimpleFileRepository>(){})
071                                .to(new TypeLiteral<Repository<Project>>(){});
072                    }
073                });
074            }
075        };
076    }
077
078    private void run() throws Exception {
079        final TrayIconApplication trayIconApplication = new TrayIconApplication(webServer, propertiesService);
080        if (trayIconApplication.isReady()) {
081            trayIconApplication.start();
082        } else {
083            System.out.println("Press Enter to stop\n");
084            //noinspection ResultOfMethodCallIgnored
085            System.in.read();
086            done();
087        }
088    }
089
090    private void done() {
091        webServer.stop();
092    }
093
094}
