016package org.maxur.perfmodel.backend.infrastructure;
017
018import org.maxur.perfmodel.backend.domain.Project;
019import org.maxur.perfmodel.backend.domain.Repository;
020import org.slf4j.Logger;
021import org.slf4j.LoggerFactory;
022
023import javax.inject.Inject;
024import java.io.File;
025import java.io.FileInputStream;
026import java.io.FileOutputStream;
027import java.io.IOException;
028import java.io.ObjectInputStream;
029import java.io.ObjectOutputStream;
030import java.util.Collection;
031import java.util.Map;
032import java.util.concurrent.ConcurrentHashMap;
033
034public class SimpleFileRepository implements Repository<Project> {
035
036    private static final Logger LOGGER = LoggerFactory.getLogger(SimpleFileRepository.class);
037
038    private static ConcurrentHashMap<String, Project> projectMap = new ConcurrentHashMap<>();
039
040    private PropertiesService propertiesService;
041
042    @Inject
043    public void setPropertiesService(PropertiesService propertiesService) {
044        this.propertiesService = propertiesService;
045        read();
046    }
047
048    @Override
049    public Project get(final String key) {
050        return projectMap.get(key);
051    }
052
053    @Override
054    public Project remove(final String key) {
055        final Project result = projectMap.remove(key);
056        write();
057        return result;
058    }
059
060    @Override
061    public Project put(final Project project) {
062        final Project result = projectMap.put(project.getName(), project);
063        write();
064        return result;
065    }
066
067    @Override
068    public Collection<Project> findAll() {
069        return projectMap.values();
070    }
071
072    private void read() {
073        final File file = new File(getFileName());
074        if (file.exists()) {
075            try (
076                    FileInputStream fileIn = new FileInputStream(file);
077                    ObjectInputStream in = new ObjectInputStream(fileIn)
078            ) {
079                //noinspection unchecked
080                projectMap = (ConcurrentHashMap<String, Project>) in.readObject();
081                LOGGER.info("Persistent Data was be restored from file " + getFileName());
082            } catch (IOException | ClassNotFoundException e) {
083                LOGGER.error("Data file is not loaded", e);
084            }
085        }
086    }
087
088    private void write() {
089        try (
090                FileOutputStream fileOut = new FileOutputStream(getFileName());
091                ObjectOutputStream out = new ObjectOutputStream(fileOut)
092        ) {
093            out.writeObject(projectMap);
094            LOGGER.info("Persistent Data was be serialized to file " + getFileName());
095        } catch (IOException e) {
096            LOGGER.error("Data file is not created", e);
097        }
098    }
099
100
101    public String getFileName() {
102        return propertiesService.asString("db.fileName", "data/projects.ser");
103    }
104}
