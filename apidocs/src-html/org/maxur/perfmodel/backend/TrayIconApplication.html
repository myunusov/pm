016package org.maxur.perfmodel.backend;
017
018import org.maxur.perfmodel.backend.infrastructure.PropertiesService;
019import org.maxur.perfmodel.backend.infrastructure.WebServer;
020import org.maxur.perfmodel.backend.rest.ApplicationResource;
021import org.slf4j.Logger;
022import org.slf4j.LoggerFactory;
023
024import java.awt.*;
025import java.net.URI;
026import java.net.URL;
027import javax.swing.*;
028
029import static org.maxur.perfmodel.backend.utils.OsUtils.isWindows;
030
031/**
032 * @author Maxim
033 * @version 1.0
034 * @since <pre>24.10.2014</pre>
035 *
036 * see http://docs.oracle.com/javase/tutorial/displayCode.html?code=http://docs.oracle.com/javase/tutorial/uiswing/examples/misc/TrayIconDemoProject/src/misc/TrayIconDemo.java
037 */
038public class TrayIconApplication {
039
040    private static final Logger LOGGER = LoggerFactory.getLogger(TrayIconApplication.class);
041
042    private final WebServer webServer;
043    private final PropertiesService propertiesService;
044    private final boolean isReady;
045
046    public TrayIconApplication(final WebServer webServer, final PropertiesService propertiesService) {
047        this.webServer = webServer;
048        this.propertiesService = propertiesService;
049        //Check the SystemTray support
050        isReady = SystemTray.isSupported();
051        if (!isReady) {
052            LOGGER.info("SystemTray is not supported");
053        } else {
054            try {
055                if (isWindows()) {
056                    UIManager.setLookAndFeel("com.sun.java.swing.plaf.windows.WindowsLookAndFeel");
057                } else {
058                    UIManager.setLookAndFeel("javax.swing.plaf.metal.MetalLookAndFeel");
059                }
060            } catch (UnsupportedLookAndFeelException | IllegalAccessException
061                    | InstantiationException | ClassNotFoundException ex
062                    ) {
063                LOGGER.error("Tray application is not created", ex);
064            }
065            UIManager.put("swing.boldMetal", Boolean.FALSE);
066        }
067    }
068
069    public boolean isReady() {
070        return isReady;
071    }
072
073    public void start() {
074        SwingUtilities.invokeLater(this::run);
075    }
076
077    private void run() {
078
079        final PopupMenu popup = new PopupMenu();
080        final TrayIcon trayIcon =
081                new TrayIcon(createImage("/img/favicon.png", "tray icon"));
082        final SystemTray tray = SystemTray.getSystemTray();
083        trayIcon.setToolTip("Performance Model Calculator");
084        trayIcon.setImageAutoSize(true);
085
086        MenuItem aboutItem = new MenuItem("About");
087        MenuItem startClientItem = new MenuItem("Start Client");
088        MenuItem manageServiceItem = new MenuItem("Stop Service");
089        MenuItem exitItem = new MenuItem("Exit");
090
091        //Add components to popup menu
092        popup.add(startClientItem);
093        popup.add(manageServiceItem);
094        popup.add(aboutItem);
095        popup.addSeparator();
096        popup.add(exitItem);
097
098        trayIcon.setPopupMenu(popup);
099
100        try {
101            tray.add(trayIcon);
102        } catch (AWTException e) {
103            LOGGER.error("TrayIcon could not be added.");
104            return;
105        }
106
107        trayIcon.addActionListener(e -> openBrowser());
108
109        startClientItem.addActionListener(e -> openBrowser());
110
111        manageServiceItem.addActionListener(e -> {
112            if (webServer.isStarted()) {
113                webServer.stop();
114            } else {
115                webServer.restart();
116            }
117            final String label = webServer.isStarted() ? "Stop Service" : "Start Service";
118            manageServiceItem.setLabel(label);
119            final String message = webServer.isStarted() ? "Web Service was started" : "Web Service was stopped";
120            trayIcon.displayMessage("Performance Model Calculator",
121                    message,
122                    TrayIcon.MessageType.INFO);
123            startClientItem.setEnabled(webServer.isStarted());
124        });
125
126        aboutItem.addActionListener(e -> JOptionPane.showMessageDialog(null,
127                "Performance Model Calculator. Version: " + ApplicationResource.VERSION));
128
129
130        exitItem.addActionListener(e -> {
131            tray.remove(trayIcon);
132            webServer.stop();
133            SystemTray.getSystemTray().remove(trayIcon);
134            System.exit(0);
135
136        });
137    }
138
139    //Obtain the image URL
140    protected static Image createImage(String path, String description) {
141        URL imageURL = TrayIconApplication.class.getResource(path);
142        if (imageURL == null) {
143            LOGGER.error("Resource not found: " + path);
144            return null;
145        } else {
146            return (new ImageIcon(imageURL, description)).getImage();
147        }
148    }
149
150    public void openBrowser() {
151        String baseUrl = propertiesService.asString("webapp.url", "http://localhost:9090/");
152        URI uri = URI.create(baseUrl);
153        Desktop desktop = Desktop.isDesktopSupported() ? Desktop.getDesktop() : null;
154        if (desktop != null && desktop.isSupported(Desktop.Action.BROWSE)) {
155            try {
156                desktop.browse(uri);
157            } catch (Exception e) {
158                e.printStackTrace();
159            }
160        }
161    }
